\def\actor{}

\newcommand{\changeactor}[1]{%
  \ifdefstrequal{\actor}{#1}%
  	{%тот же%
		\def\printactor{}
	}%
	{%
		% смена \actor{} на {#1}
		\def\actor{#1}
		\def\printactor{{\fontspec{PT Sans Bold}\actor:}}
	}
}
